3   import static de.tototec.utils.functional.FList.append;
4   import static de.tototec.utils.functional.FList.exists;

6   import org.apache.maven.plugin.logging.Log;
7   import org.apache.maven.project.MavenProject;
8   
9   public class AspectjProjectAnalyzer implements MavenProjectAnalyzer {
10  
11  	public static final String ORG_CODEHAUS_MOJO_ASPECTJ_MAVEN_PLUGIN = "org.codehaus.mojo:aspectj-maven-plugin";
12  	public static final String ORG_ECLIPSE_AJDT_CORE_AJBUILDER = "org.eclipse.ajdt.core.ajbuilder";
13  	public static final String ORG_ECLIPSE_AJDT_UI_AJNATURE = "org.eclipse.ajdt.ui.ajnature";
14  
15  	private final Log log;
16  	private final boolean addingAllowed;
17  
18  	public AspectjProjectAnalyzer(Log log, boolean addingAllowed) {
19  		this.log = log;
20  		this.addingAllowed = addingAllowed;
21  	}
22  
23  	@Override
24  	public ProjectConfig analyze(ProjectConfig projectConfig, MavenProject mavenProject) {
25  		ProjectConfig updated = projectConfig;
26  
27  		boolean pluginDetected = exists(
28  				mavenProject.getPluginArtifactMap().keySet(),
29  				key -> ORG_CODEHAUS_MOJO_ASPECTJ_MAVEN_PLUGIN.equals(key));
30  
31  		if (pluginDetected) {
32  			log.debug("Detected aspectj plugin");
33  
34  			if (addingAllowed) {
35  				log.debug("Adding scala nature and builder, disabling java nature and builder");
36  				updated = updated
37  						.withNatures(append(updated.getNatures(),
38  								new Nature(ORG_ECLIPSE_AJDT_UI_AJNATURE, "Auto-detected from pom")))
39  						.withBuilders(append(updated.getBuilders(),
40  								new Builder(ORG_ECLIPSE_AJDT_CORE_AJBUILDER, "Auto-detected from pom")))
41  						.withDisabledBuilders(append(updated.getDisabledBuilders(),
42  								JavaProjectAnalyzer.ORG_ECLIPSE_JDT_CORE_JAVABUILDER));
43  			}
44  
45  			// TODO: generate settings
46  
47  			// TODO: add aspectj classpath container (if no aspect lib is on the path)
48  		}
49  
50  		return updated;
51  	}
52  
53  }
