5   import org.apache.maven.plugin.logging.Log;
6   import org.apache.maven.project.MavenProject;

8   public class M2eProjectAnalyzer implements MavenProjectAnalyzer {
9   
10  	public static final String ORG_ECLIPSE_M2E_CORE_MAVEN2_NATURE = "org.eclipse.m2e.core.maven2Nature";
11  	public static final String ORG_ECLIPSE_M2E_CORE_MAVEN2_BUILDER = "org.eclipse.m2e.core.maven2Builder";
12  	public static final String ORG_ECLIPSE_M2E_MAVEN2_CLASSPATH_CONTAINER = "org.eclipse.m2e.MAVEN2_CLASSPATH_CONTAINER";
13  
14  	private final boolean addingAllowed;
15  
16  	public M2eProjectAnalyzer(Log log, boolean addingAllowed) {
17  		this.addingAllowed = addingAllowed;
18  	}
19  
20  	@Override
21  	public ProjectConfig analyze(final ProjectConfig projectConfig, final MavenProject mavenProject) {
22  		if (addingAllowed) {
23  			// Add M2e nature and builder
24  			return projectConfig
25  					.withBuilders(append(projectConfig.getBuilders(),
26  							new Builder(ORG_ECLIPSE_M2E_CORE_MAVEN2_BUILDER, "Default M2E Builder")))
27  					.withNatures(append(projectConfig.getNatures(),
28  							new Nature(ORG_ECLIPSE_M2E_CORE_MAVEN2_NATURE, "Defaut M2E Nature")))
29  					.withClasspathContainers(append(projectConfig.getClasspathContainers(),
30  							ORG_ECLIPSE_M2E_MAVEN2_CLASSPATH_CONTAINER));
31  		} else
32  			return projectConfig;
33  	}
