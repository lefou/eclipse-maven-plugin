19  public class ScalaProjectAnalyzer implements MavenProjectAnalyzer {
20  
21  	public static final String ORG_SCALA_IDE_SDT_CORE_SCALABUILDER = "org.scala-ide.sdt.core.scalabuilder";
22  	public static final String ORG_SCALA_IDE_SDT_CORE_SCALANATURE = "org.scala-ide.sdt.core.scalanature";
23  	public static final String ORG_SCALA_IDE_SDT_LAUNCHING_SCALA_COMPILER_CONTAINER = "org.scala-ide.sdt.launching.SCALA_COMPILER_CONTAINER";
24  
25  	private final Log log;
26  	private final boolean addingAllowed;
27  
28  	public ScalaProjectAnalyzer(Log log, boolean addingAllowed) {
29  		this.log = log;
30  		this.addingAllowed = addingAllowed;
31  	}
32  
33  	@Override
34  	public ProjectConfig analyze(ProjectConfig projectConfig, MavenProject mavenProject) {
35  		ProjectConfig updated = projectConfig;
36  
37  		final List<String> scalaPlugins = Arrays.asList(
38  				"net.alchim31.maven:scala-maven-plugin",
39  				"com.google.code.sbt-compiler-maven-plugin:sbt-compiler-maven-plugin",
40  				"com.carrotgarden.maven:scalor-maven-plugin_2.12",
41  				"com.carrotgarden.maven:scalor-maven-plugin_2.13");
42  
43  		final boolean pluginDetected = exists(
44  				mavenProject.getPluginArtifactMap().keySet(),
45  				key -> exists(
46  						scalaPlugins,
47  						p -> p.equals(key)));
48  
49  		if (pluginDetected) {
50  			log.debug("Detected scala plugin");
51  
52  			// FList.toHashMap("org.scala-ide.sdt.core.prefs", printStream -> {
53  			// TODO: generate settings file
54  
55  			// scala.compiler.installation=2.11
56  			// scala.compiler.sourceLevel=2.11
57  			// target=jvm-1.8
58  
59  			if (addingAllowed) {
60  				final Optional<String> scalaVersion = Optional
61  						.of(mavenProject.getProperties().getProperty("scala.version"))
62  						.map(v -> mkString(take(v.split("[.]"), 2), "."));
63  
64  				final Optional<String> javaVersion = projectConfig.getJavaVersion();
65  
66  				class Helper {
67  					String relativePath(String path) {
68  						return Util.relativePath(mavenProject, path);
69  					}
70  				}
71  				final Helper util = new Helper();
72  
73  				final List<String> settings = flatten(
74  						Arrays.asList(
75  								map(projectConfig.getSources(), s -> "//" + util.relativePath(s) + "=main"),
76  								map(projectConfig.getResources(), s -> "//" + util.relativePath(s.getPath()) + "=main"),
77  								map(projectConfig.getTestSources(), s -> "//" + util.relativePath(s) + "=tests"),
78  								map(projectConfig.getTestResources(),
79  										s -> "//" + util.relativePath(s.getPath()) + "=tests"),
80  								Optional.some("P="),
81  								scalaVersion.map(v -> "scala.compiler.additionalParams=\\ -Xsource\\:" + v),
82  								scalaVersion.map(v -> "scala.compiler.installation=" + v),
83  								scalaVersion.map(v -> "scala.compiler.sourceLevel=" + v),
84  								Optional.some("scala.compiler.useProjectSettings=true"),
85  								Optional.some("stopBuildOnError=true"),
86  								javaVersion.map(v -> "target=jvm-" + v),
87  								Optional.some("useScopesCompiler=true"),
88  								Optional.none()));
89  				final SettingsFile settingsFile = new SettingsFile("org.scala-ide.sdt.core.prefs", settings);
90  
91  				log.debug("Adding scala nature and builder, disabling java nature and builder");
92  				updated = updated
93  						.withNatures(prepend(
94  								new Nature(ORG_SCALA_IDE_SDT_CORE_SCALANATURE, "Auto-detected from pom"),
95  								updated.getNatures()))
96  						.withBuilders(prepend(
97  								new Builder(ORG_SCALA_IDE_SDT_CORE_SCALABUILDER, "Auto-detected from pom"),
98  								updated.getBuilders()))
99  						.withClasspathContainers(append(updated.getClasspathContainers(),
100 								ORG_SCALA_IDE_SDT_LAUNCHING_SCALA_COMPILER_CONTAINER))
101 						.withDisabledBuilders(append(updated.getDisabledBuilders(),
102 								JavaProjectAnalyzer.ORG_ECLIPSE_JDT_CORE_JAVABUILDER))
103 						.withSettingsFiles(append(updated.getSettingsFiles(), settingsFile));
104 
105 			}
106 		}
107 
108 		return updated;
109 	}
