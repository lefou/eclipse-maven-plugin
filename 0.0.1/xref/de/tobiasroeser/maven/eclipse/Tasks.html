22  	public void generateProjectFile(
23  			final PrintStream printStream,
24  			final ProjectConfig projectConfig) {
25  		printStream.println("<?xml version=\"1.0\" encoding=\"UTF-8\"?>");
26  		printStream.println("<!-- Generated by eclipse-maven-plugin -->");
27  		printStream.println("<projectDescription>");
28  		printStream.println("\t<name>" + projectConfig.getName() + "</name>");
29  		printStream.println("\t<comment>" + projectConfig.getComment() + "</comment>");
30  
31  		printStream.println("\t<projects>");
32  		printStream.println("\t</projects>");
33  
34  		printStream.println("\t<buildSpec>");
35  		foreach(projectConfig.getBuilders(), b -> {
36  			printStream.println("\t\t<buildCommand>");
37  			printStream.println("\t\t\t<name>" + b.getName() + "</name>");
38  			printStream.println("\t\t\t<arguments>");
39  			printStream.println("\t\t\t</arguments>");
40  			printStream.println("\t\t</buildCommand>");
41  		});
42  		printStream.println("\t</buildSpec>");
43  
44  		printStream.println("\t<natures>");
45  		foreach(projectConfig.getNatures(), n -> {
46  			printStream.println("\t\t<nature>" + n.getName() + "</nature>");
47  		});
48  		printStream.println("\t</natures>");
49  
50  		printStream.println("</projectDescription>");
51  	}

53  	public String relativePath(final String file) {
54  		final URI basePath = basedir.toURI();
55  		final String relPath = basePath.relativize(new File(file).toURI()).getPath();
56  		if (relPath.length() > 1 && relPath.endsWith("/")) {
57  			return relPath.substring(0, relPath.length() - 1);
58  		} else {
59  			return relPath;
60  		}
61  	}

63  	public Optional<String> whenUndefined(final Optional<?> predicate, final String useWhenDefined) {
64  		if (predicate.isDefined())
65  			return Optional.none();
66  		else
67  			return Optional.some(useWhenDefined);
68  	}

70  	public void generateClasspathFileContent(final PrintStream printStream, final ProjectConfig projectConfig,
71  			final Optional<String> buildOutput, final String outputDirectory, final String testOutputDirectory,
72  			final boolean sourcesOptional) {
73  
74  		printStream.println("<?xml version=\"1.0\" encoding=\"UTF-8\"?>");
75  		printStream.println("<!-- Generated by eclipse-maven-plugin -->");
76  		printStream.println("<classpath>");
77  
78  		// src
79  		foreach(projectConfig.getSources(),
80  				s -> generateClasspathEntry(printStream, "src", s,
81  						whenUndefined(buildOutput, outputDirectory),
82  						sourcesOptional));
83  		foreach(projectConfig.getResources(),
84  				s -> generateClasspathEntry(printStream, "src", s,
85  						whenUndefined(buildOutput, outputDirectory),
86  						sourcesOptional));
87  		foreach(projectConfig.getTestSources(),
88  				s -> generateClasspathEntry(printStream, "src", s,
89  						whenUndefined(buildOutput, testOutputDirectory),
90  						sourcesOptional));
91  		foreach(projectConfig.getTestResources(),
92  				s -> generateClasspathEntry(printStream, "src", s,
93  						whenUndefined(buildOutput, testOutputDirectory),
94  						sourcesOptional));
95  
96  		// con
97  		final String javaVersion = projectConfig.getJavaVersion();
98  		final String jrePrefix = "org.eclipse.jdt.launching.JRE_CONTAINER/org.eclipse.jdt.internal.debug.ui.launcher.StandardVMType/";
99  		final String jreSuffix;
100 		if ("1.5".equals(javaVersion) || "1.4".equals(javaVersion)) {
101 			jreSuffix = "J2SE-" + javaVersion;
102 		} else {
103 			jreSuffix = "JavaSE-" + javaVersion;
104 		}
105 
106 		generateClasspathEntry(printStream, "con", jrePrefix + jreSuffix, Optional.none(), false);
107 
108 		generateClasspathEntry(printStream, "con", "org.eclipse.m2e.MAVEN2_CLASSPATH_CONTAINER", Optional.none(),
109 				false);
110 
111 		// output
112 		printStream.println("\t<classpathentry kind=\"output\" path=\""
113 				+ buildOutput.getOrElse(relativePath(outputDirectory)) + "\"/>");
114 
115 		printStream.println("</classpath>");
116 	}
117 
118 	protected void generateClasspathEntry(
119 			final PrintStream printStream,
120 			final String kind,
121 			final String path,
122 			final Optional<String> outputPath,
123 			final boolean optional) {
124 		printStream.print("\t<classpathentry kind=\"" + kind + "\" path=\"" + relativePath(path) + "\"");
125 		outputPath.foreach(p -> printStream.print(" output=\"" + relativePath(p) + "\""));
126 		printStream.println(">");
127 		printStream.println("\t\t<attributes>");
128 		if (optional) {
129 			printStream.println("\t\t\t<attribute name=\"optional\" value=\"true\"/>");
130 		}
131 		printStream.println("\t\t\t<attribute name=\"maven.pomderived\" value=\"true\"/>");
132 		printStream.println("\t\t</attributes>");
133 		printStream.println("\t</classpathentry>");
134 	}
