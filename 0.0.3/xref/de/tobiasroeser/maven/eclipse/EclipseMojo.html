41  /**
42   * Generates Eclipse project files from the current Maven project.
43   * The typical files are:
44   * <ul>
45   * <li><code>.project</code>
46   * <li><code>.classpath</code>
47   * <li><code>.settings/org.eclipse.m2e.core.prefs</code>
48   * <li><code>.settings/org.eclipse.core.resources.prefs</code>
49   * <li><code>.settings/org.eclipse.jdt.core.prefs</code>
50   * </ul>
51   */
52  @Mojo(name = "eclipse", defaultPhase = LifecyclePhase.GENERATE_RESOURCES)
53  public class EclipseMojo extends AbstractMojo {

207 	protected ProjectConfig readPomProjectConfig() {
208 		ProjectConfig projectConfig = new ProjectConfig()
209 				.withName(Optional.lift(mavenProject.getName())
210 						.orElse(Optional.lift(mavenProject.getArtifactId()))
211 						.getOrElse("undefined"))
212 				.withComment(Optional.lift(mavenProject.getDescription()).getOrElse(""))
213 				.withSources(mavenProject.getCompileSourceRoots())
214 				.withTestSources(mavenProject.getTestCompileSourceRoots())
215 				.withResources(map(mavenProject.getBuild().getResources(), r -> r.getDirectory()))
216 				.withTestResources(map(mavenProject.getBuild().getTestResources(), r -> r.getDirectory()));
217 
218 		final Optional<String> encoding = Optional
219 				.lift(mavenProject.getProperties().getProperty("project.build.sourceEncoding"))
220 				.orElse(Optional.lift("UTF-8"));
221 		projectConfig = projectConfig.withEncoding(encoding);
222 
223 		if (defaultBuilders) {
224 			// Add M2e nature and builder
225 			projectConfig = projectConfig
226 					.withBuilders(concat(projectConfig.getBuilders(), defaultM2eBuilders()))
227 					.withNatures(concat(projectConfig.getNatures(), defaultM2eNatures()));
228 		}
229 		return projectConfig;
230 	}

232 	protected ProjectConfig readFullProjectConfig() {
233 
234 		final List<Builder> autodetectBuilders = !autodetect ? Collections.emptyList() : autodetectableBuilders();
235 		final List<Nature> autodetectNatures = !autodetect ? Collections.emptyList() : autodetectableNatures();
236 
237 		// initial config from pom
238 
239 		ProjectConfig projectConfig = readPomProjectConfig();
240 
241 		if (defaultBuilders) {
242 			projectConfig = projectConfig
243 					.withBuilders(concat(projectConfig.getBuilders(), defaultJavaBuilders()))
244 					.withNatures(concat(projectConfig.getNatures(), defaultJavaNatures()));
245 		}
246 
247 		final Optional<String> javaVersion = Optional
248 				.lift(mavenProject.getProperties().getProperty("maven.compiler.source"))
249 				.orElse(Optional.lift(mavenProject.getProperties().getProperty("maven.compiler.target")))
250 				.orElse(projectConfig.getJavaVersion());
251 		projectConfig = projectConfig.withJavaVersion(javaVersion);
252 
253 		final List<Plugin> plugins = mavenProject.getBuild().getPlugins();
254 		for (final Plugin plugin : plugins) {
255 			{
256 				// Auto-detectable Builders
257 				final List<Builder> detectedBuilders = filter(
258 						autodetectBuilders,
259 						a -> exists(a.getMavenPluginKeys(), k -> k.equals(plugin.getKey())));
260 
261 				if (!detectedBuilders.isEmpty()) {
262 					getLog().debug("Auto-detected builders [" + detectedBuilders + "] for plugin ["
263 							+ plugin.getKey() + "]");
264 				}
265 
266 				// determines, if a given builderName should be excluded
267 				final F1<String, Boolean> excludeB = builderName -> exists(
268 						detectedBuilders,
269 						b -> exists(
270 								b.getDisablesBuilders(),
271 								disablesBuilder -> {
272 									final boolean disable = builderName.equals(disablesBuilder);
273 									if (disable) {
274 										getLog().debug(
275 												"Auto-detected builder [" + b + "] disables default builder ["
276 														+ builderName + "]");
277 									}
278 									return disable;
279 								}));
280 
281 				// default Builder (if allowed) minus excluded ones plus
282 				// auto-detected ones
283 				final List<Builder> filteredBuilders = filter(projectConfig.getBuilders(),
284 						b -> !excludeB.apply(b.getName()));
285 
286 				projectConfig = projectConfig.withBuilders(concat(filteredBuilders, detectedBuilders));
287 			}
288 
289 			{
290 				// Auto-detectable Natures
291 				final List<Nature> detectedNatures = filter(
292 						autodetectNatures,
293 						a -> exists(a.getMavenPluginKeys(), k -> k.equals(plugin.getKey())));
294 
295 				if (!detectedNatures.isEmpty()) {
296 					getLog().debug("Auto-detected natures [" + detectedNatures + "] for plugin ["
297 							+ plugin.getKey() + "]");
298 				}
299 
300 				// determines, if a given natureName should be excluded
301 				final F1<String, Boolean> excludeN = natureName -> exists(
302 						detectedNatures,
303 						n -> exists(
304 								n.getDisablesNatures(),
305 								disablesNature -> {
306 									final boolean disable = natureName.equals(disablesNature);
307 									if (disable) {
308 										getLog().debug("Auto-detected nature [" + n + "] disables default nature ["
309 												+ natureName + "]");
310 									}
311 									return disable;
312 								}));
313 
314 				// default nature (if allowed) minus excluded ones plus
315 				// auto-detected ones
316 				final List<Nature> filteredNatures = filter(
317 						projectConfig.getNatures(),
318 						b -> !excludeN.apply(b.getName()));
319 
320 				projectConfig = projectConfig.withNatures(concat(filteredNatures, detectedNatures));
321 
322 			}
323 		}
324 
325 		// enhance with config values
326 		projectConfig = projectConfig
327 				.withSources(concat(projectConfig.getSources(), extraSources))
328 				.withResources(concat(projectConfig.getResources(), extraResources))
329 				.withTestSources(concat(projectConfig.getTestSources(), extraTestSources))
330 				.withTestResources(concat(projectConfig.getTestResources(), extraTestResources))
331 				.withBuilders(concat(projectConfig.getBuilders(),
332 						map(extraBuilders, b -> new Builder(b, "Explicit Builder from pom"))))
333 				.withNatures(concat(projectConfig.getNatures(),
334 						map(extraNatures, n -> new Nature(n, "Explicit Nature from pom"))));
335 		return projectConfig;
336 	}

369 	@Override
370 	public void execute() throws MojoExecutionException, MojoFailureException {
371 		if (skip) {
372 			getLog().info("Skipping eclipse");
373 			return;
374 		}
375 
376 		final File basedir = mavenProject.getBasedir();
377 		final Tasks tasks = new Tasks(basedir);
378 
379 		final String packaging = mavenProject.getPackaging();
380 
381 		final ProjectConfig projectConfig;
382 		if ("pom".equals(packaging)) {
383 			projectConfig = readPomProjectConfig();
384 		} else {
385 			projectConfig = readFullProjectConfig();
386 		}
387 
388 		// Reading/checking templates
389 		final Map<String, String> templates = new LinkedHashMap<>();
390 		// First, add templates from template dir
391 		if (settingsTemplatesDir != null && settingsTemplatesDir.exists()) {
392 			final File[] files = Optional.lift(settingsTemplatesDir.listFiles()).getOrElse(new File[0]);
393 			foreach(
394 					filter(files, f -> f.isFile()),
395 					f -> templates.put(f.getName(), f.getAbsolutePath()));
396 		}
397 		templates.putAll(settingsTemplates);
398 
399 		getLog().debug("Final eclipse project config: " + projectConfig);
400 
401 		final File projectFile = new File(basedir, ".project");
402 		generateFile(projectFile, dryrun, printStream -> {
403 			tasks.generateProjectFile(printStream, projectConfig);
404 		});
405 
406 		if (!templates.containsKey(ORG_ECLIPSE_M2E_CORE_PREFS)) {
407 			generateFile(new File(basedir, ".settings/" + ORG_ECLIPSE_M2E_CORE_PREFS), dryrun, printStream -> {
408 				tasks.generateSettingOrgEclipseM2eCorePrefs(printStream, activeProfiles);
409 			});
410 		}
411 
412 		if (!templates.containsKey(ORG_ECLIPSE_CORE_RESOURCES_PREFS)) {
413 			generateFile(new File(basedir, ".settings/" + ORG_ECLIPSE_CORE_RESOURCES_PREFS), dryrun, printStream -> {
414 				tasks.generateSettingOrgEclipseCoreResourcesPrefs(printStream, projectConfig);
415 			});
416 		}
417 
418 		if (!"pom".equals(packaging)) {
419 			final File classpathFile = new File(basedir, ".classpath");
420 			generateFile(classpathFile, dryrun, printStream -> {
421 				tasks.generateClasspathFileContent(
422 						printStream, projectConfig,
423 						Optional.lift(alternativeOutput),
424 						outputDirectory, testOutputDirectory,
425 						sourcesOptional);
426 			});
427 
428 			if (!templates.containsKey(ORG_ECLIPSE_JDT_CORE_PREFS)) {
429 				generateFile(new File(basedir, ".settings/" + ORG_ECLIPSE_JDT_CORE_PREFS), dryrun, printStream -> {
430 					tasks.generateSettingOrgEclipseJdtCorePrefs(printStream, projectConfig.getJavaVersion());
431 				});
432 			}
433 		}
434 
435 		try {
436 			foreach(templates.entrySet(), entry -> {
437 				final String fileName = ".settings/" + entry.getKey();
438 				final File templateFile = Optional.some(new File(entry.getValue()))
439 						.map(f -> f.isAbsolute() ? f : new File(basedir, f.getPath())).get();
440 				getLog().debug("Processing template file: " + templateFile);
441 				// we need to wrap the exception because our foreach-loop
442 				// cannot handle checked exceptions
443 				try {
444 					generateFile(new File(basedir, fileName), dryrun, printStream -> {
445 						try (FileReader in = new FileReader(templateFile);
446 								LineNumberReader lnr = new LineNumberReader(in);) {
447 							String line;
448 							while ((line = lnr.readLine()) != null) {
449 								printStream.println(line);
450 							}
451 						} catch (final IOException e) {
452 							throw new RuntimeException("WRAP",
453 									new MojoExecutionException("Could not read template file " + templateFile));
454 						}
455 					});
456 				} catch (final MojoExecutionException e) {
457 					throw new RuntimeException("WRAP", e);
458 				}
459 			});
460 		} catch (
461 
462 		final RuntimeException e) {
463 			if (e.getCause() instanceof MojoExecutionException) {
464 				throw (MojoExecutionException) e.getCause();
465 			} else {
466 				throw e;
467 			}
468 		}
469 	}

471 	protected void generateFile(final File file, final boolean dryrun, final Procedure1<PrintStream> generator)
472 			throws MojoExecutionException {
473 
474 		final Charset charset = StandardCharsets.UTF_8;
475 		final OutputStream out;
476 		final LinkedList<Runnable> onEnd = new LinkedList<>();
477 
478 		if (dryrun) {
479 			final ByteArrayOutputStream bout = new ByteArrayOutputStream();
480 			onEnd.add(() -> {
481 				final String fileContent = new String(bout.toByteArray(), charset);
482 				getLog().info("(dryrun) I would generate: " + file + " with content:\n" + fileContent);
483 			});
484 			out = bout;
485 		} else {
486 			if (file.exists()) {
487 				getLog().info("Overwriting existing file: " + file);
488 			} else {
489 				getLog().debug("Writing file: " + file);
490 				if (file.getParentFile() != null && !file.getParentFile().exists()) {
491 					// ensure, dir exists
492 					file.getParentFile().mkdirs();
493 				}
494 			}
495 			try {
496 				out = new FileOutputStream(file);
497 			} catch (final FileNotFoundException e) {
498 				throw new MojoExecutionException("Could not write file: " + file, e);
499 			}
500 		}
501 
502 		try (final PrintStream printStream = new PrintStream(out, false, charset.name());) {
503 			generator.apply(printStream);
504 			printStream.flush();
505 		} catch (final UnsupportedEncodingException e) {
506 			throw new MojoExecutionException("Could not create print stream", e);
507 		}
508 
509 		foreach(onEnd, r -> r.run());
510 	}
