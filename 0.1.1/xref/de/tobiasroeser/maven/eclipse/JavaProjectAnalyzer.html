1   package de.tobiasroeser.maven.eclipse;

3   import static de.tototec.utils.functional.FList.append;
4   
5   import org.apache.maven.plugin.logging.Log;
6   import org.apache.maven.project.MavenProject;
7   
8   import de.tototec.utils.functional.Optional;
9   
10  public class JavaProjectAnalyzer implements MavenProjectAnalyzer {
11  
12  	public static final String ORG_ECLIPSE_JDT_CORE_JAVABUILDER = "org.eclipse.jdt.core.javabuilder";
13  	public static final String ORG_ECLIPSE_JDT_CORE_JAVANATURE = "org.eclipse.jdt.core.javanature";
14  
15  	private final Log log;
16  	private final boolean addingAllowed;
17  
18  	public JavaProjectAnalyzer(final Log log, final boolean addingAllowed) {
19  		this.log = log;
20  		this.addingAllowed = addingAllowed;
21  	}
22  
23  	@Override
24  	public ProjectConfig analyze(final ProjectConfig projectConfig, final MavenProject mavenProject) {
25  		if ("pom".equals(mavenProject.getPackaging())) {
26  			log.debug("Skip java-specific project settings for pom project: " + mavenProject);
27  			return projectConfig;
28  		}
29  
30  		ProjectConfig updated = projectConfig;
31  		if (addingAllowed) {
32  			log.debug("Adding Java builder and nature");
33  			updated = updated
34  					.withBuilders(append(updated.getBuilders(),
35  							new Builder(ORG_ECLIPSE_JDT_CORE_JAVABUILDER, "Default Java Builder")))
36  					.withNatures(append(updated.getNatures(),
37  							new Nature(ORG_ECLIPSE_JDT_CORE_JAVANATURE, "Default Java Nature")));
38  
39  			final Optional<String> javaVersion = Optional
40  					.of(mavenProject.getProperties().getProperty("maven.compiler.source"))
41  					.orElse(Optional.of(mavenProject.getProperties().getProperty("maven.compiler.target")))
42  					.orElse(updated.getJavaVersion());
43  
44  			log.debug("Adding Java classpath container");
45  			final String cpEntry;
46  			final String jrePrefix = "org.eclipse.jdt.launching.JRE_CONTAINER";
47  			if (javaVersion.isDefined()) {
48  				final String jreMiddle = "/org.eclipse.jdt.internal.debug.ui.launcher.StandardVMType/";
49  				if ("1.5".equals(javaVersion.get()) || "1.4".equals(javaVersion.get())) {
50  					cpEntry = jrePrefix + jreMiddle + "J2SE-" + javaVersion.get();
51  				} else {
52  					cpEntry = jrePrefix + jreMiddle + "JavaSE-" + javaVersion.get();
53  				}
54  			} else {
55  				cpEntry = jrePrefix;
56  			}
57  
58  			updated = updated
59  					.withJavaVersion(javaVersion)
60  					.withClasspathContainers(append(updated.getClasspathContainers(), cpEntry));
61  
62  		}
63  		return updated;
64  	}
65  
66  }
