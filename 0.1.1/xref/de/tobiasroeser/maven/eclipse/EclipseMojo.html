40  /**
41   * Generates Eclipse project files from the current Maven project.
42   * The typical files are:
43   * <ul>
44   * <li><code>.project</code>
45   * <li><code>.classpath</code>
46   * <li><code>.settings/org.eclipse.m2e.core.prefs</code>
47   * <li><code>.settings/org.eclipse.core.resources.prefs</code>
48   * <li><code>.settings/org.eclipse.jdt.core.prefs</code>
49   * </ul>
50   */
51  @Mojo(name = "eclipse", defaultPhase = LifecyclePhase.GENERATE_RESOURCES)
52  public class EclipseMojo extends AbstractMojo {

184 	protected ProjectConfig extraConfigEnhancements(final ProjectConfig projectConfig, MavenProject mavenProject) {
185 		// enhance with config values
186 		ProjectConfig updated = projectConfig
187 				.withSources(concat(projectConfig.getSources(), extraSources))
188 				.withResources(concat(projectConfig.getResources(),
189 						map(extraResources, r -> new Resource().withPath(r))))
190 				.withTestSources(concat(projectConfig.getTestSources(), extraTestSources))
191 				.withTestResources(concat(projectConfig.getTestResources(),
192 						map(extraTestResources, r -> new Resource().withPath(r))))
193 				.withBuilders(concat(projectConfig.getBuilders(),
194 						map(extraBuilders, b -> new Builder(b, "Explicit Builder from pom"))))
195 				.withNatures(concat(projectConfig.getNatures(),
196 						map(extraNatures, n -> new Nature(n, "Explicit Nature from pom"))));
197 
198 		final File basedir = mavenProject.getBasedir();
199 
200 		// Reading/checking templates
201 		final Map<String, File> templates = new LinkedHashMap<>();
202 
203 		// First, add templates from template dir
204 		if (settingsTemplatesDir != null && settingsTemplatesDir.exists()) {
205 			final File[] files = Optional.of(settingsTemplatesDir.listFiles()).getOrElse(new File[0]);
206 			foreach(filter(files, f -> f.isFile()), f -> templates.put(f.getName(), f));
207 		}
208 
209 		// Second, add explicit settings map
210 		foreach(settingsTemplates.entrySet(), entry -> {
211 			final File templateFile = Optional.some(new File(entry.getValue()))
212 					.map(f -> f.isAbsolute() ? f : new File(basedir, f.getPath())).get();
213 			templates.put(entry.getKey(), templateFile);
214 		});
215 
216 		// Read template files
217 		for (final Entry<String, File> entry : templates.entrySet()) {
218 			getLog().debug("Processing template file: " + entry.getValue());
219 			List<String> lines;
220 			try {
221 				lines = Files.readAllLines(entry.getValue().toPath());
222 				updated = updated.withSettingsFiles(
223 						append(updated.getSettingsFiles(), new SettingsFile(entry.getKey(), lines)));
224 			} catch (final IOException e) {
225 				throw new RuntimeMojoException(
226 						new MojoExecutionException("Could not read template file: " + entry.getValue(), e));
227 			}
228 		}
229 
230 		return updated;
231 	}

233 	@Override
234 	public void execute() throws MojoExecutionException, MojoFailureException {
235 		if (skip) {
236 			getLog().info("Skipping eclipse");
237 			return;
238 		}
239 		// We handle wrapped mojo exceptions here
240 		try {
241 			final File basedir = mavenProject.getBasedir();
242 			final Tasks tasks = new Tasks(basedir, Optional.of(getLog()));
243 
244 			final String packaging = mavenProject.getPackaging();
245 
246 			final List<MavenProjectAnalyzer> analyzers = Arrays.asList(
247 					new MinimalPomAnalyzer(getLog()),
248 					new JavaProjectAnalyzer(getLog(), defaultBuilders),
249 					new ScalaProjectAnalyzer(getLog(), autodetect),
250 					new AspectjProjectAnalyzer(getLog(), autodetect),
251 					(pc, mp) -> extraConfigEnhancements(pc, mp),
252 					new M2eProjectAnalyzer(getLog(), defaultBuilders));
253 
254 			final ProjectConfig projectConfig = foldLeft(
255 					analyzers,
256 					new ProjectConfig(),
257 					(pc, a) -> a.analyze(pc, mavenProject));
258 
259 			final List<String> settingsFileNames = map(projectConfig.getSettingsFiles(), s -> s.getName());
260 
261 			getLog().debug("Final eclipse project config: " + projectConfig);
262 
263 			final File projectFile = new File(basedir, ".project");
264 			generateFile(projectFile, dryrun, printStream -> {
265 				tasks.generateProjectFile(printStream, projectConfig);
266 			});
267 
268 			if (!contains(settingsFileNames, ORG_ECLIPSE_M2E_CORE_PREFS)) {
269 				generateFile(new File(basedir, ".settings/" + ORG_ECLIPSE_M2E_CORE_PREFS), dryrun, printStream -> {
270 					tasks.generateSettingOrgEclipseM2eCorePrefs(printStream, activeProfiles);
271 				});
272 			}
273 
274 			if (!contains(settingsFileNames, ORG_ECLIPSE_CORE_RESOURCES_PREFS)) {
275 				generateFile(new File(basedir, ".settings/" + ORG_ECLIPSE_CORE_RESOURCES_PREFS), dryrun,
276 						printStream -> {
277 							tasks.generateSettingOrgEclipseCoreResourcesPrefs(printStream, projectConfig);
278 						});
279 			}
280 
281 			if (!"pom".equals(packaging)) {
282 				final File classpathFile = new File(basedir, ".classpath");
283 				generateFile(classpathFile, dryrun, printStream -> {
284 					tasks.generateClasspathFileContent(
285 							printStream, projectConfig,
286 							Optional.of(alternativeOutput),
287 							outputDirectory, testOutputDirectory,
288 							sourcesOptional);
289 				});
290 
291 				if (!contains(settingsFileNames, ORG_ECLIPSE_JDT_CORE_PREFS)) {
292 					generateFile(new File(basedir, ".settings/" + ORG_ECLIPSE_JDT_CORE_PREFS), dryrun, printStream -> {
293 						tasks.generateSettingOrgEclipseJdtCorePrefs(printStream, projectConfig.getJavaVersion());
294 					});
295 				}
296 			}
297 
298 			for (final SettingsFile settingsFile : projectConfig.getSettingsFiles()) {
299 				generateFile(new File(basedir, ".settings/" + settingsFile.getName()), dryrun, printStream -> {
300 					foreach(settingsFile.getContent(), line -> printStream.println(line));
301 				});
302 			}
303 
304 		} catch (final RuntimeMojoException e) {
305 			final Throwable cause = e.getCause();
306 			if (cause instanceof MojoExecutionException) {
307 				throw (MojoExecutionException) cause;
308 			} else if (cause instanceof MojoFailureException) {
309 				throw (MojoFailureException) cause;
310 			} else {
311 				throw e;
312 			}
313 		}
314 
315 	}

317 	protected void generateFile(final File file, final boolean dryrun, final Procedure1<PrintStream> generator)
318 			throws MojoExecutionException {
319 
320 		final Charset charset = StandardCharsets.UTF_8;
321 		final OutputStream out;
322 		final LinkedList<Runnable> onEnd = new LinkedList<>();
323 
324 		if (dryrun) {
325 			final ByteArrayOutputStream bout = new ByteArrayOutputStream();
326 			onEnd.add(() -> {
327 				final String fileContent = new String(bout.toByteArray(), charset);
328 				getLog().info("(dryrun) I would generate: " + file + " with content:\n" + fileContent);
329 			});
330 			out = bout;
331 		} else {
332 			if (file.exists()) {
333 				getLog().info("Overwriting existing file: " + file);
334 			} else {
335 				getLog().debug("Writing file: " + file);
336 				if (file.getParentFile() != null && !file.getParentFile().exists()) {
337 					// ensure, dir exists
338 					file.getParentFile().mkdirs();
339 				}
340 			}
341 			try {
342 				out = new FileOutputStream(file);
343 			} catch (final FileNotFoundException e) {
344 				throw new MojoExecutionException("Could not write file: " + file, e);
345 			}
346 		}
347 
348 		try (final PrintStream printStream = new PrintStream(out, false, charset.name());) {
349 			generator.apply(printStream);
350 			printStream.flush();
351 		} catch (final UnsupportedEncodingException e) {
352 			throw new MojoExecutionException("Could not create print stream", e);
353 		}
354 
355 		foreach(onEnd, r -> r.run());
356 	}
